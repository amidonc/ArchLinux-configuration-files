# update archlinux
    yay -Syu
	or pacman -Syu if no yay

# uncomment locale
    /etc/locale.gen
    locale-gen
   
# set default shell of new user to zsh
    vi /etc/default/useradd
    SHELL=/bin/zsh

# create new user, set password
    useradd -m -G wheel -s /bin/zsh <username>
    passwd <username>
    
# give user sudo rights
    visudo 
      Under the “User privilege specification” section, add the following line
        %wheel ALL=(ALL:ALL) ALL

# OPENSSL key
ssh-keygen -t ed25519 -C "email@example.com"
eval "$(ssh-agent -s)"
	> Agent pid 59566
ssh-add ~/.ssh/id_ed25519

	# permissions
	700	~/.shh	
	644	~/.ssh/authorized_keys
	644	~/.ssh/known_hosts
	644	~/.ssh/config
	600	~/.ssh/id_rsa
	644	~/.ssh/id_rsa.pub
          
# remove grub timeout
/etc/default/grub          
GRUB_TIMEOUT=0
grub-mkconfig -o /boot/grub/grub.cfg


# mkinitcpio -P - use this to show missing firmware

# LIBSEAT Backend
export LIBSEAT_BACKEND=logind

# Hyprland
exec dbus-run-session /usr/bin/Hyprland



###########
# OH-MY-ZSH
download & install oh-my-zsh
sh -c "$(curl -fsSL https://raw.github.com/robbyrussell/oh-my-zsh/master/tools/install.sh)"
   
download & configure powerlevel10k
git clone https://github.com/romkatv/powerlevel10k.git $ZSH_CUSTOM/themes/powerlevel10k
   
      change the value of ZSH_THEME in ~/.zshrc
      ZSH_THEME="powerlevel10k/powerlevel10k"
         
configure pl10k:
p10k configure
         
      clone plugins:
         zsh-syntax-highlighting
            git clone https://github.com/zsh-users/zsh-syntax-highlighting.git ${ZSH_CUSTOM:-~/.oh-my-zsh/custom}/plugins/zsh-syntax-highlighting
         zsh-autosuggestions
            git clone https://github.com/zsh-users/zsh-autosuggestions ${ZSH_CUSTOM:-~/.oh-my-zsh/custom}/plugins/zsh-autosuggestions
         
	activate the plugins:
            in ~/.zshrc add plugins to plugins=() line.
               plugins=( git zsh-syntax-highlighting zsh-autosuggestions)
