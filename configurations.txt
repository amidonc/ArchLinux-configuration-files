# use wsldl and rootfs.tar of latest archlinux
#   follow instructions here:
#     https://wsldl-pg.github.io/docs/How-to-Install/
      
# create wsl.conf
# https://learn.microsoft.com/en-us/windows/wsl/wsl-config
    vi /etc/wsl.conf
      [boot]
      systemd=true
    
# update archlinux
    pacman -Syu

# uncomment locale
    /etc/locale.gen
    locale-gen
   
# install zsh, zsh-autosuggestions
    pacman -S zsh zsh-autosuggestions

# set default shell of new user to zsh
    vi /etc/default/useradd
    SHELL=/bin/zsh

# create new user, set password
    useradd -m -G wheel -s /bin/zsh <username>
    passwd <username>

# give user sudo rights
    visudo 
      Under the “User privilege specification” section, add the following line
        <username> ALL=(ALL:ALL) ALL
        
# exit WSL session
 
# open powershell session
# go to archlinux installation directory
    ./archlinux config --default-user <username>

# shutdown wsl
    wsl --shutdown
